196  template <typename Container>
197  static void construct(PyObject* object, boost::python::converter::rvalue_from_python_stage1_data* data)
198  {
199  namespace python = boost::python;
200  // Object is a borrowed reference, so create a handle indicting it is
201  // borrowed for proper reference counting.
202  python::handle<> handle(python::borrowed(object));
203 
204  // Obtain a handle to the memory block that the converter has allocated
205  // for the C++ type.
206  typedef python::converter::rvalue_from_python_storage<Container> storage_type;
207  void* storage = reinterpret_cast<storage_type*>(data)->storage.bytes;
208 
209  typedef python::stl_input_iterator<typename Container::value_type> iterator;
210 
211  // Allocate the C++ type into the converter's memory block, and assign
212  // its handle to the converter's convertible variable. The C++
213  // container is populated by passing the begin and end iterators of
214  // the python object to the container's constructor.
215  new (storage) Container(iterator(python::object(handle)), // begin
216  iterator()); // end
217  data->convertible = storage;
218  }

284 class lock_context_manager
285 {
286 public:
287  lock_context_manager(LongTermMemoryConduit& ltmc) : ltmc(ltmc)
288  {
289  }
290  lock_context_manager(Lock& lock) : ltmc(lock.ltmc.get())
291  {
292  }
293 
294  // context manager protocol
295 public:
296  // Use a static member function to get a handle to the self Python
297  // object.
298  static boost::python::object enter(boost::python::object self)
299  {
300  namespace python = boost::python;
301  lock_context_manager& myself = python::extract<lock_context_manager&>(self);
302 
303  // Construct the RAII object.
304  myself.impl_ = std::make_shared<Lock>(myself.ltmc, true);
305 
306  // Return this object, allowing caller to invoke other
307  // methods exposed on this class.
308  return self;
309  }
310 
311  bool exit(boost::python::object type, boost::python::object value, boost::python::object traceback)
312  {
313  // Destroy the RAII object.
314  impl_.reset();
315  return false; // Do not suppress the exception.
316  }
317 
318  static bool acquire(boost::python::object self)
319  {
320  namespace python = boost::python;
321  lock_context_manager& myself = python::extract<lock_context_manager&>(self);
322  if (!myself.impl_)
323  {
324  myself.impl_ = std::make_shared<Lock>(myself.ltmc, true);
325  return true;
326  }
327  return false;
328  }
329 
330  static bool release(boost::python::object self)
331  {
332  namespace python = boost::python;
333  lock_context_manager& myself = python::extract<lock_context_manager&>(self);
334  if (myself.impl_)
335  {
336  // Destroy the RAII object.
337  myself.impl_.reset();
338  return true;
339  }
340  return false;
341  }
342 
343 private:
344  std::shared_ptr<Lock> impl_;
345  std::reference_wrapper<knowledge_rep::LongTermMemoryConduitInterface<LongTermMemoryConduit>> ltmc;
346 };
