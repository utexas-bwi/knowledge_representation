207  LTMCMap deepCopy(const std::string& with_name)
208  {
209  // This operation doesn't need to be fast and it doesn't need to be
210  // especially safe, so we'll do it in memory
211  LTMCMap new_map = this->ltmc.get().getMap(with_name);
212  const auto points = getAllPoints();
213  const auto poses = getAllPoses();
214  const auto regions = getAllRegions();
215  const auto doors = getAllDoors();
216  for (const auto point : points)
217  {
218  new_map.addPoint(point.getName(), point.x, point.y);
219  }
220  for (const auto pose : poses)
221  {
222  new_map.addPose(pose.getName(), pose.x, pose.y, pose.theta);
223  }
224  for (const auto region : regions)

226  new_map.addRegion(region.getName(), region.points);
227  }
228  for (const auto door : doors)

231  }
